# vite-plugin-buildInfo

## 安装
```
pnpm install vite-plugin-meta-info
```

## 使用
```ts
// vite.config.ts
import BuildInfoPlugin from 'vite-plugin-meta-info'

export default defineConfig({
    // ignore other config
    plugins: [
        // ignore other plugins
        BuildInfoPlugin()
    ]
    
})
```
